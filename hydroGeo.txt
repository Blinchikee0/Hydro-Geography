    // ---------------- Animated counters ----------------
    const counters = [

    counters.forEach(c=>{
      const el = document.getElementById(c.id);
      let start = 0; const dur = 1400; const step = Math.ceil(c.to/ (dur/30));
      const t = setInterval(()=>{
        start += step;
        if(start >= c.to){start=c.to; clearInterval(t)}
        el.textContent = start.toLocaleString();
      },30);
    });

    // ---------------- Moving ship dots over the map (DOM) ----------------
    const map = document.getElementById('map');
    const ship1 = document.createElement('div'); ship1.className='ship-dot'; ship1.style.left='6%'; ship1.style.top='62%'; map.appendChild(ship1);
    const ship2 = document.createElement('div'); ship2.className='ship-dot'; ship2.style.left='10%'; ship2.style.top='34%'; map.appendChild(ship2);

    // animate along simple bezier-ish points
    const paths = [
      [{x:6,y:62},{x:20,y:40},{x:36,y:56},{x:52,y:46},{x:68,y:36},{x:88,y:24}],
      [{x:10,y:34},{x:24,y:22},{x:40,y:32},{x:56,y:26},{x:74,y:18},{x:92,y:12}]
    ];
    function animateShip(el, pts, dur){
      let t=0; function step(){
        t+=1/60; const p = (t/dur);
        if(p>=1){t=0}
        // simple segment interpolation
        const idx = Math.floor(p*(pts.length-1));
        const frac = (p*(pts.length-1))-idx;
        const A = pts[idx], B = pts[Math.min(idx+1, pts.length-1)];
        const x = A.x + (B.x-A.x)*frac; const y = A.y + (B.y-A.y)*frac;
        el.style.left = x + '%'; el.style.top = y + '%';
        requestAnimationFrame(step);
      }
      step();
    }
    animateShip(ship1, paths[0], 12);
    animateShip(ship2, paths[1], 18);

    // ---------------- Floating decorative bubbles ----------------
    document.querySelectorAll('.bubble').forEach((b,i)=>{
      const delay = i*0.8; b.animate([
        {transform:'translateY(0) scale(1)'},

        {transform:'translateY(0) scale(1)'}
      ],{duration:5000+ i*800, iterations:Infinity, easing:'ease-in-out', delay:delay*1000});
